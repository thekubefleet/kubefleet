Get Started in Minutes

# Clone the repository
git clone https://github.com/thekubefleet/kubefleet.git
cd kubefleet
# Start the dashboard
go run ./cmd/server
# Start the agent
go run ./cmd/agent
# Open dashboard
open http://localhost:3000
